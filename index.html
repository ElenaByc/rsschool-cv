public class Fibonacci {
    public static long fib (int n) {
        if (n == 0) {
            return 0;
        }
        long curr = 1;
        long prev = 1;
        for (int i = 3; i <= n; i++) {
            curr = curr + prev;
            prev = curr - prev;
        }
        return curr;
    }
}
